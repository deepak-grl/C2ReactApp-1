import GrlPdApiLib
import GrlPdLib
class Program
{
static void Main(string[] args)
{

//Detach the controller before configuring
objController.Detach();
//Set sink capability of controller as 5V /3A, 12V /3A, 15V/ 3A, 20V/ 3A

snkCap.AddFixedSupply((uint)(9000 / 50), (uint) (3000 / 10));
snkCap.AddFixedSupply((uint)(12000 / 50), (uint) (3000 / 10));
snkCap.AddFixedSupply((uint)(15000 / 50), (uint) (3000 / 10));

objController.SetSinkCapability(snkCap);
//Set Request (PDO#1, 2A), after the PD contract controller send a request for PDO1 with 2A current
RequestMessage reqMsg = new RequestMessage();
reqMsg.AddFixedVariableRequestDataObj((uint)(3000 / 10), (uint) (1000 / 10));
objController.SetSinkRequestMessage(reqMsg);
objController.SetControllerMode(ControllerMode.Sink);
//start capture and show plot
List<MeasurementChannel> channels = new List<MeasurementChannel>();
channels.Add(MeasurementChannel.PORTA_VBUS_CURRENT);
channels.Add(MeasurementChannel.PORTA_VBUS_VOLTAGE);
obj.ShowPlot(channels);

}

int GrlPdApiLib.GrlC2Controller.Attach ( ControllerPort  port = ControllerPort.ONE)

Enables the GRL-USB-PD-C2/F1 controller to be programmatically controlled when plugged in with the DUT. Depending on the controller mode, this function presents Rp/Rd from the CC line and changes the state of the controller to Source Attach or Sink Attach. It is recommended to call the Detach function before configuring the controller mode, setting the source sink and source capabilities and calling this Attach function.


Python Example:

# Refer Python environment setup section to set up Python environment and how to import the packages before following this example
#import the packages
import GrlPdLib
import GrlPdApiLib
#Create instance of the object
objController = GrlPdApiLib.GrlC2Controller()
retVal = objController.Initialize("192.168.255.1")
#set the controller mode as Source
objController.SetControllerMode(GrlPdLib.ControllerMode.Source);
# set the cable type
objController.UseTesterCable(null, GrlPdLib.ControllerPort.ONE)
# Configure Source Capability
srcCap = GrlPdLib.SourceCapability()
# Configure PDO1 as 5V 3A
srcCap.AddFixedSupply(5000/50, 3000/10)
# Configure PDO2 as 9V 3A
srcCap.AddFixedSupply(9000/50, 3000/10)
# Configure PDO3 as 12V 3A
srcCap.AddFixedSupply(12000/50, 3000/10)
# Configure Source Capability for Port 1 of the controller
objController.SetSourceCapability(srcCap, ControllerPort.ONE)
#Attach the conroller
objController.Attach(GrlPdLib.ControllerPort.ONE)

  1. This parameter defines the sub sampling rate at which the signal channels are getting captured. Maximum supported sample rate by C2 controller in 2.4 Million Samples/second for 1 channel, if multiple channels are configured, then sample rate will get reduced by those many number of channels configured
  2. Recommended default value for this parameter in 1000, which means that configured signal lines will be captured at 2400 Samples/second for 1 channel
